procedure compass {
    read north;
    read south;
    west = 100;
    east = 50;

    if (north/2 = 0) then {
        south = north + 2;
    } else {
        west = east + 50;
    }

    while (north < south) {
        if (east < west) then {
            east = east + 20;
        } else {
            west = east + 20;
        }
        while (north > east) {
            north = east * 10 - west;
        }
        north = north + east - west;
    }

    central = north * 10 + south * 100 - east * 10 + west * 20;
    print central;
    print north;
}
